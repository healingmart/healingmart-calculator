(() => {
  document.getElementById("hm-calculator-self-test-host")?.remove();

  try {
    delete window.HMCalculatorSelfTest;
  } catch (e) {
    window.HMCalculatorSelfTest = undefined;
  }

  [...document.scripts]
    .filter(s => s.src && s.src.includes("hm-calculator-self-test"))
    .forEach(s => s.remove());

  const s = document.createElement("script");
  s.src =
    "https://healingmart.github.io/healingmart-calculator/test/" +
    "hm-calculator-self-test.v1.0.1.js?v=" +
    Date.now();

  document.head.appendChild(s);
})();
